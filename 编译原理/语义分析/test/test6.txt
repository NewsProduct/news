//if，for，while语句的嵌套，选择choic；若choic==1，返回N以内素数和；若choice==2，返回N内数和；
main()
{
  int i,N;
  int sum = 0;int choice=0;
  choice=read();
  N = read();
  if(choice == 1) {
      for(i=1;i<=N;i=i+1)
      {
         if(i%2 == 1)
	    sum = sum+i;
      }
  }
  else{
      i=0;
      while(i<N){
      sum = sum + i;
      i = i + 2;
     }
   }
  write(sum);
}
